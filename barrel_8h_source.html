111void Brew::validateBrewInstallation() {
112 std::string const check_path =
113 install_path_ + LE_SPACER + getCommandHead(BrewCommandType::Builtin::VERSION);
114
115 BarrelCmd::Proc proc(check_path, BarrelCmd::Stream::STDOUT_STDERR);
116 proc.execute();
117
118 if (proc.getExitStatus() == EXIT_SUCCESS) {
119 Brew::is_installed = true;
120 std::string const dump = proc.getStreamDump();
121 install_version_ = {dump.begin(), std::find(dump.begin(), dump.end(), '\n')};
122 } else {
123 throw std::runtime_error(
124 "Brew::validateBrewInstallation(): Homebrew installation failed to validate!");
125 }
126}

199BrewCommand<E>::BrewCommand(Brew const& brew, E cmd, Args... args)
200 : cmd_(cmd), head_(getCommandHead(cmd)), chain_(brew.getInstallPath()) {
201 (q_.push(std::forward<Args>(args)), ...);
202 chain_ += LE_SPACER + head_;
203
204 while (!q_.empty()) {
205 std::visit(
206 [&chain_ = chain_](auto&& arg) {
207 using Tt = std::decay_t<decltype(arg)>;
208 if constexpr (std::is_same_v<Tt, const char*> || std::is_same_v<Tt, std::string>)
209 chain_ += LE_SPACER + arg;
210 // Consider string_view too
211 else
212 static_assert(FalseType<Tt>::value,
213 "BrewCommand(Brew const&, E, Args...): Barrel Dev Error! Fix the "
214 "if-constexpr to make it exhaustive wrt the types q_ supports");
215 },
216 q_.front());
217 q_.pop();
218 }
219};
