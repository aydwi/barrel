105void Brew::validateBrewInstallation() {
106 std::string const check_path =
107 install_path_ + LE_SPACER + getBrewCommandHead(BrewCommandType::Builtin::VERSION);
108
109 BarrelCmd::Proc proc(check_path, BarrelCmd::Stream::STDOUT_STDERR);
110 proc.execute();
111
112 if (proc.getExitStatus() == EXIT_SUCCESS) {
113 Brew::is_installed = true;
114 std::string const dump = proc.getStreamDump();
115 install_version_ = {dump.begin(), std::find(dump.begin(), dump.end(), '\n')};
116 } else {
117 throw std::runtime_error(
118 "Brew::validateBrewInstallation(): Homebrew installation failed to validate!");
119 }
120}
